  runApp(MyApp());                                                 // ‡¶Ö‡ßç‡¶Ø‡¶æ‡¶™ ‡¶ö‡¶æ‡¶≤‡ßÅ

class MyApp extends StatefulWidget {
  @override
  State<MyApp> createState() => _MyAppState();
}

class _MyAppState extends State<MyApp> {
  final _formKey = GlobalKey<FormState>();                         // Form ‡¶è‡¶∞ Key (Validation ‡¶è‡¶∞ ‡¶ú‡¶®‡ßç‡¶Ø)
  final TextEditingController _nameCtrl = TextEditingController(); // Name Input ‡¶è‡¶∞ ‡¶ú‡¶®‡ßç‡¶Ø Controller
  final TextEditingController _passCtrl = TextEditingController(); // Password Input ‡¶è‡¶∞ ‡¶ú‡¶®‡ßç‡¶Ø Controller

  String? savedName;                                               // ‡¶∏‡ßá‡¶≠ ‡¶ï‡¶∞‡¶æ ‡¶®‡¶æ‡¶Æ ‡¶¶‡ßá‡¶ñ‡¶æ‡¶®‡ßã‡¶∞ ‡¶ú‡¶®‡ßç‡¶Ø
  bool isLoggedIn = false;                                         // ‡¶≤‡¶ó‡¶á‡¶® ‡¶∏‡ßç‡¶ü‡ßç‡¶Ø‡¶æ‡¶ü‡¶æ‡¶∏

  @override
  void initState() {
    super.initState();
    loadUserData();                                                // ‡¶Ö‡ßç‡¶Ø‡¶æ‡¶™ ‡¶ö‡¶æ‡¶≤‡ßÅ‡¶∞ ‡¶∏‡¶Æ‡ßü ‡¶°‡ßá‡¶ü‡¶æ ‡¶≤‡ßã‡¶°
  }

  // üîπ SharedPreferences ‡¶•‡ßá‡¶ï‡ßá ‡¶°‡ßá‡¶ü‡¶æ ‡¶≤‡ßã‡¶°
  void loadUserData() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    setState(() {
      savedName = prefs.getString('username');                     // ‡¶®‡¶æ‡¶Æ ‡¶≤‡ßã‡¶°
      isLoggedIn = prefs.getBool('isLoggedIn') ?? false;           // ‡¶≤‡¶ó‡¶á‡¶® ‡¶∏‡ßç‡¶ü‡ßç‡¶Ø‡¶æ‡¶ü‡¶æ‡¶∏ ‡¶≤‡ßã‡¶°
    });
  }

  // üîπ ‡¶≤‡¶ó‡¶á‡¶® ‡¶ï‡¶∞‡ßá ‡¶°‡ßá‡¶ü‡¶æ ‡¶∏‡ßá‡¶≠
  void loginUser() async {
    SharedPreferences prefs = await SharedPreferences.getInstance();
    await prefs.setString('username', _nameCtrl.text);             // ‡¶®‡¶æ‡¶Æ ‡¶∏‡ßá‡¶≠
    await prefs.setBool('isLoggedIn', true);                       // ‡¶≤‡¶ó‡¶á‡¶® ‡¶∏‡ßç‡¶ü‡ßç‡¶Ø‡¶æ‡¶ü‡¶æ‡¶∏ ‡¶∏‡ßá‡¶≠
    setState(() {

  void logoutUser() async {

    setState(() {

                      child: Text("Logout"),

                      mainAxisAlignment: MainAxisAlignment.center,
                      children: [
                        TextFormField(                             // ‡¶®‡¶æ‡¶Æ ‡¶á‡¶®‡¶™‡ßÅ‡¶ü
                          controller: _nameCtrl,
                          decoration: InputDecoration(
                              labelText: "Enter your name",

                            if (value == null || value.isEmpty) {
                              return 'Please enter your name';
                            }
                            return null;
                          },
                        ),
                        SizedBox(height: 20),
                        TextFormField(                             // ‡¶™‡¶æ‡¶∏‡¶ì‡ßü‡¶æ‡¶∞‡ßç‡¶° ‡¶á‡¶®‡¶™‡ßÅ‡¶ü

                        SizedBox(height: 20),
                        ElevatedButton(
                          onPressed: () {

                              loginUser();                         // ‡¶°‡ßá‡¶ü‡¶æ ‡¶∏‡ßá‡¶≠ ‡¶ì ‡¶≤‡¶ó‡¶á‡¶®
